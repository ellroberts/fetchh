    ]},
    { label: 'Status', items: [

      { name: '--color-border-form', value: '#e9ecef', note: 'form' },
    ]},
  ]

  const Swatch = ({ t }: { t: { name: string, value: string, note: string } }) => {
    const light = parseInt(t.value.slice(1,3),16)/255 * 0.2126 + parseInt(t.value.slice(3,5),16)/255 * 0.7152 + parseInt(t.value.slice(5,7),16)/255 * 0.0722 > 0.4
    const fg = light ? '#231F1A' : '#FAF8F5'
    return (
      <div onClick={() => copy(t.name)} style={{ backgroundColor: t.value, border: `1px solid ${light ? 'rgba(0,0,0,0.08)' : 'rgba(255,255,255,0.1)'}`, borderRadius: 6, padding: '10px 12px', cursor: 'pointer', minHeight: 70, display: 'flex', flexDirection: 'column', gap: 4, position: 'relative', userSelect: 'none' }}>

        <span style={{ fontSize: 10, color: fg, fontFamily: 'monospace', marginTop: 'auto', opacity: 0.85 }}>{t.value}</span>

      </div>
    )
  }

            {ramp.steps.map(t => <div key={t.name} style={{ flex: 1, backgroundColor: t.value }} title={t.note} />)}
          </div>
          {/* Swatches */}
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(9, 1fr)', gap: 'var(--spacing-2)' }}>
            {ramp.steps.map(t => <Swatch key={t.name} t={t} />)}
          </div>
        </div>
      ))}

      {/* Divider */}
      <div style={{ borderTop: '1px solid var(--color-warm-300)', paddingTop: 'var(--spacing-6)' }}>

          </div>
        ))}
      </div>
    </div>
  )
